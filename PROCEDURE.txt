SET SERVEROUTPUT ON
CREATE OR REPLACE PROCEDURE SHOW(M IN NUMBER,N OUT NUMBER,S OUT NUMBER)
IS
CURSOR S1 IS SELECT ENAME,EMPNO,SAL FROM EMP WHERE MGR=M;
BEGIN
SELECT COUNT (EMPNO),SUM(SAL)INTO N,S FROM EMP WHERE MGR=M;
DBMS_OUTPUT.PUT_LINE(M);
FOR I IN S1
LOOP
DBMS_OUTPUT.PUT_LINE(I.EMPNO||'   '||I.ENAME);
END LOOP;
END;
/
